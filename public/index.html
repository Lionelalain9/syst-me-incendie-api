async function sInscrire(){ const nom=document.getElementById('reg-nom').value.trim(); const email=document.getElementById('reg-email').value.trim(); const mdp=document.getElementById('reg-mdp').value; const mdp2=document.getElementById('reg-mdp2').value; const btn=document.getElementById('btn-register'); const msg=document.getElementById('msg-register'); if(!nom||!email||!mdp){showMsg(msg,'Veuillez remplir tous les champs.',false);return;} if(mdp!==mdp2){showMsg(msg,'Les mots de passe ne correspondent pas.',false);return;} if(mdp.length<8){showMsg(msg,'Mot de passe trop court (min. 8).',false);return;} btn.disabled=true;btn.textContent='Création...'; try{ const r=await fetch(`${API}/api/auth/inscription`,{ method:'POST', headers:{'Content-Type':'application/json'}, body:JSON.stringify({nom,email,mot_de_passe:mdp}) }); const d=await r.json(); if(!r.ok){showMsg(msg,d.erreur||'Erreur.',false);return;} token=d.token; userData=d.user; localStorage.setItem('sdi_token',token); localStorage.setItem('sdi_user',JSON.stringify(userData)); showMsg(msg,'✅ Compte créé avec succès !',true); setTimeout(afficherDashboard,800); } catch(e){showMsg(msg,'Serveur inaccessible.',false);} finally{btn.disabled=false;btn.textContent='✅ Créer mon compte';} }
